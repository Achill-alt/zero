```python
import json


def parse_json_field(value):
    """Parse a JSON field that may already be a dict/list or a JSON string.

    Used for SQLAlchemy Text columns storing JSON (conditions, steps, step_results).
    """
    if isinstance(value, str):
        return json.loads(value)
    return value
```

```python
from sqlalchemy.orm import Session
from sqlalchemy import text


def search_contracts(db: Session, q: str = "", contract_type: str = None, status: str = None,
                     date_from: str = None, date_to: str = None, page: int = 1, page_size: int = 10):
    """Full-text search using FTS5 with optional filters. Falls back to LIKE for CJK queries."""
    clauses = []
    params = {}

    if q:
        safe_q = q.replace('"', '""')
        fts5_clause = "c.id IN (SELECT rowid FROM contracts_fts WHERE contracts_fts MATCH :fts_query)"
        clauses.append(fts5_clause)
        params["fts_query"] = f'"{safe_q}"'
        use_fts5 = True
    else:
        clauses.append("1=1")
        use_fts5 = False

    if contract_type:
        clauses.append("c.contract_type = :contract_type")
        params["contract_type"] = contract_type

    if status:
        clauses.append("c.status = :status")
        params["status"] = status

    if date_from:
        clauses.append("c.created_at >= :date_from")
        params["date_from"] = date_from

    if date_to:
        clauses.append("c.created_at <= :date_to")
        params["date_to"] = date_to

    where = " AND ".join(clauses)

    highlight_cols = ""
    if use_fts5:
        highlight_cols = """,
            snippet(contracts_fts, 0, '<mark>', '</mark>', '...', 32) as title_highlight,
            snippet(contracts_fts, 1, '<mark>', '</mark>', '...', 64) as content_highlight
            FROM contracts c
            JOIN contracts_fts fts ON c.id = fts.rowid
        """
    else:
        highlight_cols = """,
            NULL as title_highlight, NULL as content_highlight
            FROM contracts c
        """

    base_sql = f"""
        SELECT c.id, c.title, c.content, c.contract_type, c.amount, c.party_a, c.party_b,
               c.start_date, c.end_date, c.status, c.creator_id, c.created_at, c.updated_at
               {highlight_cols}
        WHERE {where}
    """

    # Count total
    count_sql = f"SELECT COUNT(*) as cnt FROM contracts c WHERE {where}"
    # For FTS5 count, join is needed
    if use_fts5:
        count_sql = f"""
            SELECT COUNT(*) as cnt FROM contracts c
            JOIN contracts_fts fts ON c.id = fts.rowid
            WHERE {where}
        """
    total = db.execute(text(count_sql), params).scalar()

    # If FTS5 returned no results, fall back to LIKE for CJK support
    if total == 0 and q:
        clauses = ["(c.title LIKE :like_q OR c.content LIKE :like_q)"]
        params = {"like_q": f"%{safe_q}%"}

        if contract_type:
            clauses.append("c.contract_type = :contract_type")
            params["contract_type"] = contract_type
        if status:
            clauses.append("c.status = :status")
            params["status"] = status
        if date_from:
            clauses.append("c.created_at >= :date_from")
            params["date_from"] = date_from
        if date_to:
            clauses.append("c.created_at <= :date_to")
            params["date_to"] = date_to

        where = " AND ".join(clauses)
        base_sql = f"""
            SELECT c.id, c.title, c.content, c.contract_type, c.amount, c.party_a, c.party_b,
                   c.start_date, c.end_date, c.status, c.creator_id, c.created_at, c.updated_at,
                   NULL as title_highlight, NULL as content_highlight
            FROM contracts c
            WHERE {where}
        """
        count_sql = f"SELECT COUNT(*) as cnt FROM contracts c WHERE {where}"
        total = db.execute(text(count_sql), params).scalar()

    # Paginate
    paginated_sql = base_sql + " ORDER BY c.created_at DESC LIMIT :limit OFFSET :offset"
    params["limit"] = page_size
    params["offset"] = (page - 1) * page_size

    rows = db.execute(text(paginated_sql), params).fetchall()

    items = []
    for row in rows:
        item = {
            "id": row.id, "title": row.title, "content": row.content,
            "contract_type": row.contract_type, "amount": row.amount,
            "party_a": row.party_a, "party_b": row.party_b,
            "start_date": str(row.start_date) if row.start_date else None,
            "end_date": str(row.end_date) if row.end_date else None,
            "status": row.status, "creator_id": row.creator_id,
            "created_at": str(row.created_at) if row.created_at else None,
            "updated_at": str(row.updated_at) if row.updated_at else None,
        }
        if q:
            item["title_highlight"] = row.title_highlight
            item["content_highlight"] = row.content_highlight
        items.append(item)

    return {"items": items, "total": total, "page": page, "page_size": page_size}
```

```python
def get_expiring_contracts(db: Session, days: int = 30, page: int = 1, page_size: int = 20):
    from datetime import date, timedelta

    today = date.today()
    cutoff = today + timedelta(days=days)

    query = (
        db.query(Contract)
        .filter(
            Contract.status.in_(["approved", "archived", "pending_approval"]),
            Contract.end_date <= cutoff,
        )
        .order_by(Contract.end_date.asc())
    )

    total = query.count()
    items = query.offset((page - 1) * page_size).limit(page_size).all()

    result_items = []
    for c in items:
        days_left = (c.end_date - today).days
        result_items.append({
            "contract": c,
            "days_left": days_left,
            "expired": days_left < 0,
        })

    return {"items": result_items, "total": total, "page": page, "page_size": page_size}
```

```python
def get_pending_approvals(db: Session, user: User, page: int = 1, page_size: int = 20):
    """Get contracts pending approval for the current user's approver_role."""
    if not user.approver_role:
        return {"items": [], "total": 0, "page": page, "page_size": page_size}

    instances = (
        db.query(ApprovalInstance)
        .filter(ApprovalInstance.status == "in_progress")
        .all()
    )

    result = []
    for inst in instances:
        template = db.query(ApprovalChainTemplate).filter(ApprovalChainTemplate.id == inst.template_id).first()
        if not template:
            continue

        steps = parse_json_field(template.steps)
        if inst.current_step_index >= len(steps):
            continue

        current_step = steps[inst.current_step_index]
        if current_step.get("role") == user.approver_role:
            contract = db.query(Contract).filter(Contract.id == inst.contract_id).first()
            if contract:
                result.append({
                    "instance": inst,
                    "contract": contract,
                    "current_step": current_step,
                    "template_name": template.name,
                })

    total = len(result)
    start = (page - 1) * page_size
    items = result[start:start + page_size]

    return {"items": items, "total": total, "page": page, "page_size": page_size}
```
